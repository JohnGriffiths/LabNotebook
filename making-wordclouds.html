pdfs_list = glob.glob('Recovery_papers_for_word_cloud/*.pdf')

def get_text(fname):
  pdf = PyPDF2.PdfFileReader(open(fname, "rb"))
  text = ' '.join([page.extractText().replace('\n', ' ') for page in pdf.pages])
  return text  

texts = {p: get_text(p) for p in pdfs_list}
